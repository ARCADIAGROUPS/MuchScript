# Print

FUN add(num1, num2) -> num1 + num2

# List of names
WOW names = ["Jose", "Maria", "John", "Mary"]
WOW len_of_names = LEN(names)

# Prints all the names in the list
# We get object from list by /num not [num]
FOR i = 0 TO len_of_names THEN
	PRINT("Name: " + names/i)
END
